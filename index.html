Input Processing

# Neural Network Inputs (Normalized)
def get_inputs(bird, pipes):
    # Bird Y position (0-1 range)
    bird_y_norm = bird.y / WINDOW_HEIGHT
    
    # Distance to pipe gap center
    pipe_center = pipes[0].height + pipes[0].GAP / 2
    distance_to_center = (bird.y - pipe_center) / 400
    
    # Velocity (normalized)
    velocity_norm = bird.velocity / 20.0
    
    # Horizontal distance to pipe
    horizontal_distance = (pipes[0].x - bird.x) / 600
    
    return [bird_y_norm, distance_to_center, 
            velocity_norm, horizontal_distance]
